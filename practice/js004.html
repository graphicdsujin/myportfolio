$(function() {
       var slide_img = setInterval(Slide, 3000);
      function Slide() {
          var img = $('.photo_wrap02 .thumb').eq(1).data('img');
          $('.photo_wrap02 .thumb').removeClass('on');
          $('.photo_wrap02 .thumb').eq(1).addClass('on')
          $('#photo_wrap_img').attr('src', img);
          $(".photo_wrap02 .thumb:first-child").insertAfter(".photo_wrap02 .thumb:last-child");
       };
 
        $('.photo_wrap02 .thumb').mouseenter(function(e) {
          var img = $(this).data('img');
          $('.photo_wrap02 .thumb').removeClass('on');
          $(this).addClass('on')
          $('#photo_wrap_img').attr('src', img);
          clearInterval(slide_img);
        });
 
        $('.photo_wrap02 .thumb').mouseleave(function(e) {
          slide_img = setInterval(Slide, 3000);
        });
      });
